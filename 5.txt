import java.util.Scanner;

public class Chatbot {
    public static void main(String[] args) {
        Scanner sc=new Scanner(System.in);

        System.out.println("Welcome to the ChatBot");
        String input;

        while(true){
            System.out.println("You: ");
            input=sc.nextLine().toLowerCase();

            if(input.equals("hello") || input.equals("hi")){
                System.out.println("Bot: How can I help you?");
            } else if (input.contains("product")) {
                System.out.println("Bot: We provide mobile,laptops and other accessories.");
            } else if (input.contains("price")) {
                System.out.println("Bot: The price starts from 10,000.");
            } else if (input.contains("delivery")) {
                System.out.println("Bot: Delivery is done at 3-4 working days.");
            } else if (input.contains("contact")) {
                System.out.println("Bot: You can contact at support@gmail.com.");
            } else if (input.contains("thank")) {
                System.out.println("Bot: you're welcome!");
            } else if (input.contains("bye")) {
                System.out.println("Bot: Thank you for visiting!");
                break;
            }else {
                System.out.println("Bot: Sorry, I am not able to understand.");
            }
        }
        sc.close();
    }
}
