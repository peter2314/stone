// BFS

import java.util.LinkedList;
import java.util.Queue;
import java.util.Scanner;

public class BFSGraph {

    private int ver;
    private LinkedList<Integer>[] adjList;

    BFSGraph(int v) {
        ver = v;
        adjList = new LinkedList[ver];

        for (int i = 0; i < v; i++) {
            adjList[i] = new LinkedList<>();
        }
    }

    void addEdge(int v, int w) {
        adjList[v].add(w);
        adjList[w].add(v);
    }

    void BFS(int start) {

        System.out.println("BFS Traversal................");

        boolean[] vis = new boolean[ver];

        Queue<Integer> queue = new LinkedList<>();

        vis[start] = true;

        queue.add(start);

        while (!queue.isEmpty()) {

            int v = queue.poll();

            System.out.print(v + " ");

            for (int i : adjList[v]) {

                if (!vis[i]) {

                    vis[i] = true;

                    queue.add(i);
                }
            }
        }

        System.out.println();
    }

    public static void main(String[] args) {

        Scanner sc = new Scanner(System.in);

        System.out.println("Enter the no of vertices");
        int v = sc.nextInt();

        System.out.println("Enter the number of edges");
        int e = sc.nextInt();

        BFSGraph g = new BFSGraph(v);

        System.out.println("Enter the starting and ending vertex of the edge");

        for (int i = 0; i < e; i++) {

            int v1 = sc.nextInt();
            int v2 = sc.nextInt();

            g.addEdge(v1, v2);
        }

        System.out.println("Enter the starting vertex");
        int s = sc.nextInt();

        g.BFS(s);
        sc.close();
    }
}

// DFS
import java.util.LinkedList;
import java.util.Scanner;

public class DFSGraph {

    private int ver;
    private LinkedList<Integer>[] adjList;

    DFSGraph(int v) {
        ver = v;
        adjList = new LinkedList[ver];

        for (int i = 0; i < v; i++) {
            adjList[i] = new LinkedList<>();
        }
    }

    void addEdge(int v, int w) {
        adjList[v].add(w);
        adjList[w].add(v);
    }

    void DFS(int start) {
        boolean[] vis = new boolean[ver];

        System.out.println("DFS Traversal................");
        dfsRecursive(start, vis);

        System.out.println();
    }

    void dfsRecursive(int s, boolean[] vis) {
        vis[s] = true;

        System.out.print(s + " ");

        for (int i : adjList[s]) {
            if (!vis[i]) {
                dfsRecursive(i, vis);
            }
        }
    }

    public static void main(String[] args) {

        Scanner sc = new Scanner(System.in);

        System.out.println("Enter the no of vertices");
        int v = sc.nextInt();

        System.out.println("Enter the number of edges");
        int e = sc.nextInt();

        DFSGraph g = new DFSGraph(v);

        System.out.println("Enter the starting and ending vertex of the edge");

        for (int i = 0; i < e; i++) {
            int v1 = sc.nextInt();
            int v2 = sc.nextInt();

            g.addEdge(v1, v2);
        }

        System.out.println("Enter the starting vertex");
        int s = sc.nextInt();

        g.DFS(s);
        sc.close();
    }
}
