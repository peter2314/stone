🪜 Steps to Run in Salesforce
Go to Developer Console → File → New → Apex Class
Name it Calculator1 → Paste Apex code → Save
Go to File → New → Visualforce Page
Name it CalculatorPage → Paste VF code → Save
Click Preview button at the top of the editor.

//Calculator1
public class Calculator1 {

    public Decimal x {get; set;}
    public Decimal y {get; set;}
    public Decimal result {get; set;}

    public void doAdd() {
        result = x + y;
    }

    public void doSubtract() {
        result = x - y;
    }

    public void doMultiply() {
        result = x * y;
    }

    public void doDivide() {
        if (y != 0) {
            result = x / y;
        } else {
            result = 0;
            ApexPages.addMessage(
                new ApexPages.Message(ApexPages.Severity.ERROR, 'Cannot divide by zero')
            );
        }
    }
}

//CalculatorPage
<apex:page controller="Calculator1">
    <apex:form>

        <h2>Calculator Application</h2>

        <apex:pageMessages />

        Enter First Number:
        <apex:inputText value="{!x}" />
        <br/><br/>

        Enter Second Number:
        <apex:inputText value="{!y}" />
        <br/><br/>

        <apex:commandButton value="Add" action="{!doAdd}" />
        <apex:commandButton value="Subtract" action="{!doSubtract}" />
        <apex:commandButton value="Multiply" action="{!doMultiply}" />
        <apex:commandButton value="Divide" action="{!doDivide}" />

        <br/><br/>

        <b>Result = {!result}</b>

    </apex:form>
</apex:page>
