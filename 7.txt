public class Sorting {
public static void main() {
Integer[] arr= new Integer[]{5, 3, 8, 1, 2};
Integer n = arr.size();
for (Integer i=0; i < n - 1; i++) { 
for (Integer j= 0; j < n - i - 1 ; j++) { if (arr[j] > arr[j + 1]) 
{ 
// swap arr[j] & arr[j+1] 
Integer temp = arr[j];
 arr[j] = arr[j+1];
 arr[j+1]= temp; }
}
}
System.debug('Sorted numbers:'+ arr);
}
}

Label: Customer
Object Name: Customer
Check Allow Reports and Allow Activities
Save

2)Add Fields
* Create fields like:

Phone__c (Phone)
Email__c (Email)
Address__c (Text Area)

3) Go to Developer Console → File → New → Apex Class

public class CustomerService {

    // Getter method for Visualforce
    public List<Customer__c> getCustomers() {
        return [SELECT Id, Name, Phone__c, Email__c FROM Customer__c];
    }

    // Insert customer
    public void addCustomer(String name, String phone, String email) {
        Customer__c c = new Customer__c();
        c.Name = name;
        c.Phone__c = phone;
        c.Email__c = email;
        insert c;
    }

    // Delete customer
    public void deleteCustomer(Id custId) {
        Customer__c c = [SELECT Id FROM Customer__c WHERE Id = :custId];
        delete c;
    }
}

4) Developer Console → New → Apex Trigger

trigger CustomerTrigger on Customer__c (after insert) {
    for(Customer__c c : Trigger.new) {

    }
}
